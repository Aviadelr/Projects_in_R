# Helper function: 
wordle_match <- function(guess, word)  # 1: correct location, -1: wrong location, 0: missing
{
  L <- nchar(guess)
  match <- rep(0, L)
  for(i in 1:L)
  {
    if(grepl(substr(guess, i, i), word, fixed=TRUE))
      {match[i] = -1}
    if(substr(guess, i, i) == substr(word, i, i))
    {      match[i] = 1}
  }
  
  return(match)
}

freq_per_chapter<-function(query,chapters=chapters,chap_words=chap_words){

q4c_fun = function(vec_of_words, table_of_probs){# MLE estimator by multiplying all letters prob
  df_of_mle = data.frame(matrix(ncol=2))
  colnames(df_of_mle)=c("word", "MLE")
  for(i in 1:length(vec_of_words)){
      splitted_word = c(unlist(strsplit(vec_of_words[i], '')))
      likelihood_1 = letter_probs$letter_prob_1[letter_probs$Letter==splitted_word[1]]
      likelihood_2 = letter_probs$letter_prob_2[letter_probs$Letter==splitted_word[2]]
      likelihood_3 = letter_probs$letter_prob_3[letter_probs$Letter==splitted_word[3]]
      likelihood_4 = letter_probs$letter_prob_4[letter_probs$Letter==splitted_word[4]]
      likelihood_5 = letter_probs$letter_prob_5[letter_probs$Letter==splitted_word[5]]
      MLE = likelihood_1 * likelihood_2 * likelihood_3 * likelihood_4 * likelihood_5
      df_of_mle = rbind(df_of_mle, c(vec_of_words[i], MLE))
  }
  df_of_mle=drop_na(df_of_mle)
  return(df_of_mle)
}

the number of turns needed is based on attempt for 5 letters, each one is an independent variable of the geometric distribution. In order to calculate the success of each variable we will use the probability of succeeding on the number i attempt minus the cumulative probability of all the attempts beforehend(i-1 attempts). Since we need all 5 variables to succeed, we will raise the said formula by the power of 5.

#Here is the actual calculation
expected_value = 0
value = 0
vec_of_values = c()
#q for each variable is 25/26
#cumulative for each attempt is 1-q^i
for (i in 1:10000) {
  formula = ((1-(25/26)^i)^5 - (1-(25/26)^(i-1))^5)
  expected_value = expected_value + (i * formula)
  value = value + formula
  vec_of_values[i] = value
}
